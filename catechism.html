ក្រុមអ្នកណែនាំអប់រំជំនឿ (មនុស្សពេញវ័យ)

ក- សកម្មភាព និងការទទួលខុសត្រូវរបស់ក្រុមគ្រូធំ

១/. រចនាសម្ព័ន្ធគ្រូអប់រំជំនឿមនុស្សពេញវ័យ

១- កញ្ញា ម៉ៅ សាវេត ប្រធាន

២- លោកស្រី ចាន់ សេរី អនុប្រធាន

៣- កញ្ញា ម៉ៅ សាវឿន សមាជិក

៤- បងប្រុស សុខ ធឿន សមាជិក

៥- កញ្ញា ជា សុវណ្ណច័ន្ទបុប្ផា លេខា

២/. ថ្នាក់អប់រំជំនឿមនុស្សពេញវ័យ

១- ថ្នាក់ស្វាគមន៍ ក្រុមត្រៀមសុំចូលព្រះសហគមន៍ (ទទួលជំហានទី១) ប្រើសៀវភៅ ខ្ញុំជាផ្លូវ + ឯកសារផ្សេងពាក់ព័ន្ធនឹងព្រះសហគមន៍។
ណែនាំដោយ អ្នកគ្រូ សាវឿន អ្នកគ្រូ សេរី និងលោកពុក

២- ថ្នាក់ជំហានទី១ ប្រើសៀវភៅ ខ្ញុំជាសេចក្តីពិត និងខ្ញុំជាផ្លូវ+ ឯកសារផ្សេងៗពាក់ព័ន្ធនឹងព្រះសហគមន៍
ណែនាំដោយ បងប្រុស ធឿន

៣- ថ្នាក់ត្រៀមជំហានទី២ នៃអគ្គសញ្ញាជ្រមុជទឹក ប្រើសៀវភៅខ្ញុំជាជីវិត + ឯកសារផ្សេងៗពាក់ព័ន្ធនឹងព្រះសហគមន៍។
ណែនាំដោយ អ្នកគ្រូ បុប្ផា និងអ្នកគ្រូ សាវេត

៤- ថ្នាក់ត្រៀមលាបថ្ងាស
ក្រុមពន្លកថ្មី ដកស្រង់ចេញពីព្រះគម្ពីរ និងឯកសារផ្សេងៗពាក់ព័ន្ធនឹងព្រះសហគមន៍
ណែនាំដោយ អ្នកគ្រូ សាវេត

៣/. ការប្រើសៀវភៅសម្រាប់អប់រំជំនឿមនុស្សពេញវ័យ

១- សៀវភៅ ខ្ញុំជាផ្លូវ

២- សៀវភៅ ខ្ញុំជាសេចក្តីពិត

៣- សៀវភៅ ខ្ញុំជាជីវិត

៤- ដកស្រង់ចេញពីព្រះគម្ពីរ និងឯកសារផ្សេងៗពាក់ព័ន្ធនឹងព្រះសហគមន៍

៤/. ក្នុងឆ្នាំ២០២៣

+ មានបេក្ខជនសុំចូលព្រះសហគមន៍ (ជំហានទី១) ចំនួន ១៦នាក់

+ មានបេក្ខជនទទួលអគ្គសញ្ញាជ្រមុជទឹក ចំនួន ០៧នាក់

+ មានបេក្ខជនទទួលអគ្គសញ្ញាលាបថ្ងាសសរុបមណ្ឌលខាងជើង+បរទេស ចំនួន ៥៩នាក់។

៥/. សកម្មភាពផ្សេងៗ

+ បានរៀបចំមេរៀនសម្រាប់ប្រចាំឆ្នាំអប់រំជំនឿ

+ បានរួមចំណែកតូចមួយក្នុងពិធីតាំងលោកបូជាចារ្យខ្មែរ៣អង្គ

+ បានចូលរួមទទួលស្វាគមន៍លោកបូជាចារ្យខ្មែរ៤អង្គ មកអភិបូជាលើកដំបូងនៅព្រះសហគមន៍យើង

+ បានចូលរួមប្រជុំជាមួយក្រុមប្រឹក្សាព្រះសហគមន៍ ដឹកនាំដោយលោកពុក មួយខែម្តង ពិភាក្សា កិច្ចការមួយចំនួនក្នុងព្រះសហគមន៍

+ បានចូលរួមសំណាក់ធម៌ក្រុមគ្រូអប់រំជំនឿទាំងអស់

+ បានចូលរួមរៀនបំប៉នបន្ថែមសម្រាប់គ្រូអប់រំជំនឿជាមួយបងស្រី និងលោកពុក

+ បានរៀបចំកម្មវិធីសំណាក់ធម៌សម្រាប់អ្នកស្វែងយល់ទទួលពិធីសុំចូលព្រះសហគមន៍ (ជំហានទី១) និងត្រៀមពិធីអគ្គសញ្ញាលាបថ្ងាស

+ បានរៀបចំក្រុមអានអត្ថបទសម្រាប់អភិបូជារៀងរាល់ថ្ងៃអាទិត្យ

+ បានចូលរួមសិក្ខាសាលាគ្រូមណ្ឌលសកម្មភាពខាងជើង ធម៌ប្រកាសជំនឿ

+ បានចូលរួមសិក្ខាលាសាគ្រូអប់រំជំនឿអន្តរភូមិភាគនៅព្រះសហគមន៍សន្តប៉ូល សន្តសិលា (ភ្នំពេញថ្មី)

+ បានរៀបចំសិស្សសម្រាប់ជួយលើកតង្វាយថ្ងៃអាទិត្យទី១, អាទិត្យទី៣ និងអាទិត្យទី៥ តាមវេនសិស្សអប់រំជំនឿ

+ បានរៀបចំធម៌អង្វរសកលសម្រាប់អាទិត្យទី៥

+ បានចូលរួមសម្របសម្រួលជាបងស្រី សម្រាប់ក្រុមរិះគិតការត្រាស់ហៅ របស់គ្រួសារបងស្រីព្រះហឬទ័យ

+ បានចូលរួមបិទឆ្នាំអប់រំជំនឿ ឆ្នាំសិក្សា ២០២២-២០២៣។

ខ- ទិសដៅបន្តឆ្នាំ២០២៤

+បន្តថ្នាក់អប់រំជំនឿមនុស្សពេញវ័យ រៀងរាល់ថ្ងៃអាទិត្យ និងថ្ងៃធម្មតា

- អ្នកគ្រូ សាវឿន អប់រំពេលព្រឹក ថ្នាក់ស្វាគមន៍ និង ត្រៀមទទួលជំហានទី១ ប្រើសៀវភៅខ្ញុំជាផ្លូវ + ឯកសារផ្សេងៗពាក់ព័ន្ធនឹងព្រះសហគមន៍

- បងប្រុស ធឿន អប់រំពេលព្រឹក ថ្នាក់ជំហានទី១ ប្រើសៀវភៅ ខ្ញុំជាផ្លូវ + ខ្ញុំជាសេចក្តីពិត + ឯកសារផ្សេងៗពាក់ព័ន្ធនឹងព្រះសហគមន៍ និងមានអប់រំថ្ងៃអង្គារ និងថ្ងៃពុធ សម្រាប់អ្នកត្រូវទទួលអគ្គសញ្ញាជ្រមុជទឹក +លាបថ្ងាស ផងដែរ។

- អ្នកគ្រូ សេរី និងលោកពុក អប់រំពេលរសៀល ថ្នាក់ស្វាគមន៍ និង ត្រៀមទទួលជំហានទី១ ប្រើសៀវភៅខ្ញុំជាផ្លូវ + ឯកសារផ្សេងៗពាក់ព័ន្ធនឹងព្រះសហគមន៍។

- អ្នកគ្រូ បុប្ផា អប់រំពេលព្រឹក + រសៀល ថ្នាក់ត្រៀមជំហានទី២ នៃអគ្គសញ្ញាជ្រមុជទឹក ប្រើសៀវភៅខ្ញុំជាជីវិត + ឯកសារផ្សេងៗពាក់ព័ន្ធនឹងព្រះសហគមន៍។

- អ្នកគ្រូ សាវេត អប់រំរៀងរាល់ថ្ងៃពុធ ថ្នាក់ត្រៀមជំហានទី២ នៃអគ្គសញ្ញាជ្រមុជទឹក ប្រើសៀវភៅខ្ញុំជាជីវិត និងឯកសារផ្សេងៗពាក់ព័ន្ធនឹងព្រះសហគមន៍។

+ គ្រោងនឹងមានបេក្ខជន សុំចូលព្រះសហគមន៍ (ជំហានទី១)

+ គ្រោងរៀបចំ បេក្ខជន ធ្វើពិធីត្រាស់ហៅយ៉ាងឱឡារិក (ជំហានទី២)

+ គ្រោងរៀបចំ បេក្ខជន ទទួលអគ្គសញ្ញាជ្រមុជទឹក

+ គ្រោងរៀបចំ បេក្ខជន ទទួលអគ្គសញ្ញាលាបថ្ងាស

+ គ្រោងមានសំណាក់ធម៌ក្រុមគ្រូទាំងអស់

+ បន្តជំរុញបងប្អូនដែលមកស្វែងយល់ព្រះបន្ទូល ឱ្យចូលរួមអភិបូជា និងមានសកម្មភាពខ្លះៗក្នុងព្រះសហគមន៍

+ បន្តចូលរួមប្រជុំជាមួយក្រុមគ្រូទាំងអស់ មួយខែម្តង ដឹកនាំដោយលោកពុក ដើម្បីពិភាក្សា និងរកដំណោះស្រាយបញ្ហាផ្សេងៗតាមថ្នាក់នីមួយៗ

+ បន្តចូលរួមសកម្មភាពផ្សេងៗរបស់ព្រះសហគមន៍ដែលបានគ្រោងរៀបចំ។

ក្រុមគ្រូអប់រំជំនឿ (គ្រូតូច)

• កូនសិស្សសរុបទាំងអស់ 116 នាក់ ចែកចេញជា 7 ថ្នាក់៖

 ថ្នាក់ដំបូង-ថ្នាក់កូនក្ងាន

 ថ្នាក់ដំបូង-សិស្សនិយាយភាសាវៀតណាម

 ថ្នាក់ត្រៀមទទួលព្រះកាយ

 ថ្នាក់ទទួលព្រះកាយ

 ថ្នាក់ចម្រើនព្រះបន្ទូល (ត្រៀមលាបថ្ងាស)

 ថ្នាក់ចម្រើនជំនឿ (ត្រៀមលាបថ្ងាស)

 ថ្នាក់លាបថ្ងាស។

សកម្មភាពក្នុងរយៈពេលមួយឆ្នាំ៖

• គ្រូតាមថ្នាក់នីមួយៗបានរៀបចំមេរៀនសម្រាប់បង្រៀនកូនសិស្សយ៉ាងយកចិត្តទុកដាក់
